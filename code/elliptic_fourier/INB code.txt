double range_coord = max_coord - min_coord
double range_found = max_found - min_found

// found * mult = coord
double found_to_coord_multiplier = range_coord / range_found;


// centre minimums around 0
for(auto &coord: result)
{
	// centre minimums around 0
	result[0] -= min_found;
	result[1] -= min_found;
	// multiply
	result[0] *= found_to_coord_multiplier;
	result[1] *= found_to_coord_multiplier;
	// correct minimums
	result[0] += min_coord;
	result[1] += min_coord;
}
